lua{
local function filterItemsByDiscountOrMarkdown(items)
    local filteredItems = {}
    for _, item in ipairs(items) do
        if item.Discount or item.Markdown then
            table.insert(filteredItems, item)
        end
    end
    return filteredItems
end

return filterItemsByDiscountOrMarkdown(wf.vars.parsedCsv)
}lua
